C + Python + GDB

module.c


#include "module.h"

int foo(int x, int y)
{
  x += 1;
  y += 2;

  return x + y;
}

module.h


#ifndef MODULE_H
#define MODULE_H

int foo(int x, int y);

#endif

driver.py


import ctypes

def main():
    module = ctypes.CDLL('./libmodule.so')
    module.foo.restype = int

    z = module.foo(1, 2)
    print(z)

if __name__ == '__main__':
    main()

Make a shared library to debug


gcc -c -Wall -Werror -fpic -ggdb module.c
gcc -shared -o libmodule.so module.o

Run the program as normal


$ python driver.py
=> 6
